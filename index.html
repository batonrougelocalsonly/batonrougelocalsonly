<script>

// === CONFIG: your Google Doc ID ===

const docUrl = "https://docs.google.com/document/d/1faFqQT7y4IW7WFoxw52Q2YZYsk8De-MpweFBt3vcmww/export?format=html";


const player = document.getElementById("player");

const playlistEl = document.getElementById("playlist");

let tracks = [];

let currentIndex = 0;


// Initialize Wavesurfer

const wave = WaveSurfer.create({

  container: '#waveform',

  waveColor: '#888',

  progressColor: '#f90',

  height: 80,

  responsive: true

});


// Fetch Google Doc and parse MP3 links

fetch(docUrl)

  .then(res => res.text())

  .then(html => {

    const parser = new DOMParser();

    const doc = parser.parseFromString(html, "text/html");

    const links = Array.from(doc.querySelectorAll("a[href$='.mp3']"));

    tracks = links.map(link => ({

      title: link.textContent || link.href.split('/').pop(),

      src: link.href

    }));

    buildPlaylist();

    loadTrack(0);

  });


// Build playlist DOM

function buildPlaylist() {

  playlistEl.innerHTML = "";

  tracks.forEach((track, index) => {

    const li = document.createElement("li");

    li.textContent = track.title;

    li.addEventListener("click", ()=>loadTrack(index));

    playlistEl.appendChild(li);

  });

}


// Highlight active track

function setActiveTrack(index) {

  Array.from(playlistEl.children).forEach((li,i)=>li.classList.toggle("active", i===index));

}


// Load track

function loadTrack(index) {

  currentIndex = index;

  player.src = tracks[index].src;

  player.play();

  setActiveTrack(index);

  wave.load(tracks[index].src);

}


// Next / Previous

function nextTrack() { loadTrack((currentIndex+1)%tracks.length); }

function prevTrack() { loadTrack((currentIndex-1+tracks.length)%tracks.length); }


document.getElementById("next").addEventListener("click", nextTrack);

document.getElementById("prev").addEventListener("click", prevTrack);


// Autoplay next track

player.addEventListener("ended", nextTrack);


// Sync Wavesurfer with audio element

player.addEventListener("play", ()=>wave.play());

player.addEventListener("pause", ()=>wave.pause());

player.addEventListener("seeked", ()=>wave.seekTo(player.currentTime/player.duration));

player.addEventListener("timeupdate", ()=>wave.setProgress(player.currentTime/player.duration));

player.addEventListener("ended", ()=>wave.stop());


</script>
